 16 function Engine(canvas, context, gameType) {
 17   if (gameType === "2D") {
 18     this.game_type = "2D";
 19     this.canvas = canvas;
 20     this.context = context;
 21     this.updateHandler = null;
 22     this.drawHandler = null;
 23     this.init();
 24   }
 25   else {
 26     this.game_type = "3D";
 27     this.updateHandler = null;
 28     this.drawHandler = null;
 29     // TODO : change naming of this function to accomodate both 2d and 3d
 30     // Parameters : webglCanvas, inputTrianglesURL, inputSpheresURL
 31     // Also for 3D games, there is no init yet. Everything initializd in Graphics Constructor
 32     this.webglCanvas = canvas;
 33     this.inputTriangles = context;
 34     this.inputSpheres = gameType;
 35     //console.log("Engine constructor :"+ this.inputTriangles);
 36     this.init();
 37   }
 38 }
 39 /**
 40  * This is the constructor function for the Engine Class
 41  */
 42 Engine.prototype.init = function () {
 43   if (this.game_type == "2D") {
 44     this.objects = new Array();           //Game objects(sprites)
 45     this.objectIdGenerator = 0;           //Unique id for each game object.
 46     this.particleSystem = new Array();   // Particle System
 47 
 48     this.input = new Input(this);         //Input handler system
 49     this.collision = new Collision(this); //Collision handler system
 50     this.storage = new Storage(this);
 51     this.physics = new Physics(this);
 52     this.network = new Network(this);
 53     this.graph = new Graph(this);
 54     this.pathSearch = new PathSearch(this);
 55 
 56     //Bind engine event listeners
 57     this.canvas.onmousedown = this.input.handleMouseDown.bind(this.input);
 58     this.canvas.onmouseup = this.input.handleMouseUp.bind(this.input);
 59     this.canvas.onmousemove = this.input.handleMouseMove.bind(this.input);
 60     document.addEventListener("keydown", this.input.handleKeyPress.bind(this.input));
 61   }
 62   else {
 63     this.input = new Input(this);         //Input handler system
 64     this.textures = new Textures(this);
 65     this.graphics = new Graphics(this);
 66     this.sound = new Sound(this);
 67 
 68     document.addEventListener("keydown", this.input.handleKeyPress.bind(this.input));
 69   }
 70 }

121   var index = this.getObjectIndex(id);

264  */
265 Collision.prototype.checkCollision = function (object, x, y) {
266   var minX = object.X;
267   var maxX = object.X + object.width;
268   var minY = object.Y;
269   var maxY = object.Y + object.height;
270 
271   if (x >= minX && x <= maxX && y >= minY && y <= maxY)
272     return true;
273   return false;
274 }

283 /**
284  * Check if a point x, y collides with any of the game objects
285  */
286 Collision.prototype.checkCollisionWithAllObjects = function (x, y) {
287   for (var i = 0; i < this.engine.objects.length; i++) {
288     if (this.checkCollision(this.engine.objects[i], x, y)) {
289       return true;
290     }
291   }
292   return false;

295 /**
296  * Check if the last moved Object interests with any other object. 
297  */
298 Collision.prototype.update = function () {
299   if (this.movedObjectId == null || Object.keys(this.movedObjectId).length == 0)
300     return;
301 
302   for (var id in this.movedObjectId) {
303     var movedObject = this.engine.getObject(id);
304 
305     if (movedObject == null)
306       return;
307 
308     var movedX = movedObject.X + (movedObject.width / 2);
309     var movedY = movedObject.Y + (movedObject.height / 2);
310 
311     for (var i = 0; i < this.engine.objects.length; i++) {
312 
313       if (this.isMovingObject(this.engine.objects[i].id) == false || this.checkCollisionBetweenMovingObjects == true) {
314         if (id != this.engine.objects[i].id && this.checkCollision(this.engine.objects[i], movedX, movedY)) {
315           if (this.collisionHandler != null) {
316             this.collisionHandler(movedObject, this.engine.objects[i]);
317             break;
318           }
319         }
320       }
321     }
322     //this.movedObjectId = null;
323   }
324 
325   this.movedObjectId = new Object();
326 }
327 
328 /**
329  * Check if the given object id belongs to a moving object
330  */
331 Collision.prototype.isMovingObject = function (id) {
332   return this.movedObjectId != null && this.movedObjectId[id] == true;
333 }

408 /**
409  * call handler if an object was clicked on.
410  * Checks for collision, If it has occurred, determines difference between mouse postion and position of the object 
411  * and calls the event handler of the game
412  */
413 Input.prototype.handleMouseDown = function (e) {
414   var x = e.clientX, y = e.clientY;
415   for (var iter = 0; iter < this.engine.objects.length; iter++) {
416     if (this.engine.collision.checkCollision(this.engine.objects[iter], x, y)) {
417       this.intervalX = x - this.engine.objects[iter].X;
418       this.intervalY = y - this.engine.objects[iter].Y;
419       this.setObjectSelected(this.engine.objects[iter].id);
420       if (this.mouseDownHandler != null)
421         this.mouseDownHandler(this.engine.objects[iter], x - this.intervalX, y - this.intervalY);
422       break;
423     }
424   }
425 }

510  * The Main Physics class constructor

514 function Physics(engine) {
515   this.engine = engine;
516   this.outOfBoundsHandler = null;

519 /**
520  * Sets Out of bounds handler
521  * 
522  * @param {any} handler Event Handler function
523  */
524 Physics.prototype.setOutOfBoundsHandler = function (handler) {
525   this.outOfBoundsHandler = handler
526 }
527 /**
528  * The Update function for physics sub engine
529  * 
530  */
531 Physics.prototype.update = function () {
532   for (var i = 0; i < this.engine.objects.length; i++) {
533     if (this.engine.objects[i].physics) {
534       var obj = this.engine.objects[i];
535       if (obj.physics.x_velocity != 0 || obj.physics.y_velocity != 0) {
536         this.engine.input.setMovedObject(obj.id);
537         obj.X += obj.physics.x_velocity;
538         obj.Y += obj.physics.y_velocity;
539         if (obj.X < 0 || (obj.X + 1.5 * obj.width) > this.engine.canvas.width) {
540           if (this.outOfBoundsHandler != null)
541             this.outOfBoundsHandler("X", obj);
542         }
543         if (obj.Y < 0 || (obj.Y + obj.height) > this.engine.canvas.height)
544           if (this.outOfBoundsHandler != null)
545             this.outOfBoundsHandler("Y", obj);
546       }
547     }
548   }
549 }

623 /**
624  * to add texture to GL
625  * 
626  * @param {any} texture 
627  */
628 Textures.prototype.handleLoadedTexture = function (texture) {
629   gl.bindTexture(gl.TEXTURE_2D, texture);
630   gl.pixelStorei(gl.UNPACK_FLIP_Y_WEBGL, true);
631   gl.texImage2D(gl.TEXTURE_2D, 0, gl.RGBA, gl.RGBA, gl.UNSIGNED_BYTE, texture.image);
632   gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_MIN_FILTER, gl.LINEAR);
633   gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_WRAP_S, gl.CLAMP_TO_EDGE);
634   gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_WRAP_T, gl.CLAMP_TO_EDGE);
635   gl.bindTexture(gl.TEXTURE_2D, null);
636 }

858 /**
859  * Get Shortest Path between a given source and dest
860  * 
861  * @param {any} graph 
862  * @param {any} source 
863  * @param {any} dest 
864  * @returns 
865  */
866 PathSearch.prototype.astar = function(graph,source, dest){
867   this.nodes = graph.nodes;
868   this.edges = graph.edges; 
869   this.closedList = new Set();
870   this.openList = new Set();
871   this.distance = new Map();
872   this.total = new Map();
873   this.predecessors = new Map();
874   this.distance[source] = 0;
875   this.total[source] = graph.heuristic(source, dest); 
876   this.openList.add(source);
877   while(this.openList.size>0){
878     var node = this.getMinimum(this.openList);
879     if(node == dest)
880       break;
881     this.closedList.add(node);
882     this.openList.delete(node);
883     adjacentNodes = this.getNeighbors(node);
884     for(var i = 0; i< adjacentNodes.length ;i++){
885       if(this.getShortestTotal(adjacentNodes[i])>this.getShortestTotal(node)+this.getDistance(node, adjacentNodes[i])){
886         this.distance[adjacentNodes[i]] = this.getShortestTotal(node)+this.getDistance(node, adjacentNodes[i]);
887         this.total[adjacentNodes[i]] = this.distance[adjacentNodes[i]] + graph.heuristic(adjacentNodes[i],node);
888         this.predecessors[adjacentNodes[i]] = node;
889         this.openList.add(adjacentNodes[i]);
890       }
891     } 
892   }
893   return this.getPath(dest);
894 }
895 /**
896  * Returns List of Neighbors
897  * 
898  * @param {any} node given node in the graph
899  * @returns list of nodes
900  */
901 PathSearch.prototype.getNeighbors = function(node){
902   var neighbors = new Array();
903   for(var i = 0; i < this.edges.length; i++){
904     if(this.edges[i].source == node && !this.closedList.has(this.edges[i].goal)){
905       neighbors.push(this.edges[i].goal);
906     }
907   }
908 
909   return neighbors;
910 }
911 /**
912  * Gets the distance from node to target
913  * 
914  * @param {any} node 
915  * @param {any} target 
916  * @returns 
917  */
918 PathSearch.prototype.getDistance = function(node, target){
919   for(var i = 0; i < this.edges.length;i++){
920     if(this.edges[i].source == node && this.edges[i].goal == target)
921       return this.edges[i].weight;
922   }
923 }
924 /**
925  * Return the path given the target
926  * 
927  * @param {any} target target of the search
928  * @returns path List of Nodes
929  */
930 PathSearch.prototype.getPath = function(target){
931   var path = new Array();
932   var step = target;
933   if(this.predecessors[step] == null){
934     return null;
935   }
936   path.push(step);
937   while(this.predecessors[step]!=null){
938     step = this.predecessors[step];
939     path.push(step);
940   }
941 
942   return path;
943   
944 }

954 /**
955  * Selects minimum Valued node from OpenList
956  * 
957  * @param {any} vertexes OpenList
958  * @returns minimum cost Node
959  */
960 PathSearch.prototype.getMinimum = function(vertexes){
961   minimum = null;
962   for(let i of vertexes){
963     if(minimum == null) {
964       minimum = i;
965     }
966     else{
967       if(this.getShortestTotal(i)<this.getShortestTotal(minimum))
968         minimum = i;
969     }
970   }
971   return minimum;
972 }
